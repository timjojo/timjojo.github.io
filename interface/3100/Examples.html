//========================================================
// Sample program for the Interface analog input 
// boards/cards
// under Windows 7/Server 2008/Vista/XP/Server 2003/
// 2000/NT/Me/98/95
//   Console application
//   (Run the program on the MS-DOS prompt or  
//    command prompt.)
//
//   Copyright 2000, 2011 Interface Corporation. 
//   All rights reserved.
//
// Description
//   How to input analog data.
//========================================================
      
#include <windows.h>
#include <stdio.h>
#include <FbiAd.h>
    
void main(void)    
{
HANDLE    hDeviceHandle; // Device handle
WORD    wSmplData[512][2]; // Sampling data (for 12 and 16-bit board/card)
// BYTE data (for 8-bit board/card)
// DWORD data (for 24-bit board/card)
ADSMPLREQ SmplConfig; // Sampling conditions
ULONG ulSmplNum; // Number of samples
// Initialize the FBIAD1.
hDeviceHandle = AdOpen("FBIAD1");
if (hDeviceHandle == INVALID_HANDLE_VALUE) {
printf("FBIAD1 is not available.\n");
exit(0);
}
      
// Retrieve the sampling conditions.
AdGetSamplingConfig(hDeviceHandle, &SmplConfig);
     
// Set 512 to the number of samples.
// Sampling channels are channel 1 and channel 2.
SmplConfig.ulSmplNum = 512;
SmplConfig.ulChCount = 2;
SmplConfig.SmplChReq[0].ulChNo = 1;
SmplConfig.SmplChReq[1].ulChNo = 2;
AdSetSamplingConfig(hDeviceHandle, &SmplConfig);
       
// Start the sampling as a non-overlapped operation.
AdStartSampling(hDeviceHandle, FLAG_SYNC);
     
// Retrieve the sampling data.
ulSmplNum = 512;
AdGetSamplingData(hDeviceHandle, &wSmplData[0][0], &ulSmplNum);
      
// Clean up and exit.
AdClose(hDeviceHandle);
}

 

//========================================================
// Sample program for the Interface analog input 
// boards/cards
// under Windows 7/Server 2008/Vista/XP/Server 2003/ 
// 2000/NT/Me/98/95
//   Console application
//   (Run the program on the MS-DOS prompt or  
//    command prompt.)
//
//   Copyright 2000, 2011 Interface Corporation. 
//   All rights reserved.
//
// Description
// The infinite sampling is performed asynchronously. // An event is occurred by a multiple of 1024 and 
// retrieve the data.
//  And save the retrieved data to the file.
//  Therefore, stop the sampling when writing 10 times //  to the file.
//========================================================
      
#include <windows.h>
#include <stdio.h>
#include <FbiAd.h>
    
void main(void)    
{
int nRet;
HANDLE    hDeviceHandle; // Device handle
WORD    wSmplData[1024][2]; // Sampling data (for 12 and 16-bit board/card)
// BYTE data (for 8-bit board/card)
// DWORD data (for 24-bit board/card)
ADSMPLREQ SmplConfig; // Sampling conditions
ULONG ulSmplNum; // Number of samples
HANDLE hEvent; // Event handle
FILE *fp; // File handle
      
// Initialize the FBIAD1.
hDeviceHandle = AdOpen("FBIAD1");
if (hDeviceHandle == INVALID_HANDLE_VALUE) {
printf("FBIAD1 is not available.\n");
exit(0);
}
      
// Open the file.
if ((fp = fopen("data.dat", "w+b")) == NULL){
  printf("Failed to open the file.\n");
  exit(0);
}
   
hEvent = CreateEvent(0, TRUE, FALSE, NULL);
nRet = AdSetBoardConfig(hDeviceHandle, hEvent, NULL, 0);
if(nRet != AD_ERROR_SUCCESS){
  printf("Error AdSetBoardConfig : %08x\n", nRet);
  CloseHandle(hEvent);
  AdClose(hDeviceHandle);
  exit(0);
}
   
// Retrieve the sampling conditions.
nRet = AdGetSamplingConfig(hDeviceHandle, &SmplConfig);
if(nRet != AD_ERROR_SUCCESS){
  printf("Error AdGetSamplingConfig : %08x\n", nRet);
  CloseHandle(hEvent);
  AdClose(hDeviceHandle);
  exit(0);
}
      
// Set 2048 to the number of samples.
// Set 1024 to the number of events.
// Set infinite sampling to trigger mode.
// Channel 1 and channel 2 is set to the
// sampling channel.
SmplConfig.ulSmplNum = 2048;
SmplConfig.ulSmplEventNum = 1024;
SmplConfig.ulTrigMode = AD_ETERNITY;
SmplConfig.ulChCount = 2;
SmplConfig.SmplChReq[0].ulChNo = 1;
SmplConfig.SmplChReq[0].ulRange = AD_5V;
SmplConfig.SmplChReq[1].ulChNo = 2;
SmplConfig.SmplChReq[1].ulRange = AD_5V;
AdSetSamplingConfig(hDeviceHandle, &SmplConfig);
if(nRet != AD_ERROR_SUCCESS){
  printf("Error AdSetSamplingConfig : %08x\n", nRet);
  CloseHandle(hEvent);
  AdClose(hDeviceHandle);
  exit(0);
}
   
// Start the sampling as an overlapped operation.
AdStartSampling(hDeviceHandle, FLAG_ASYNC);
if(nRet != AD_ERROR_SUCCESS){
  printf("Error AdStartSampling : %08x\n", nRet);
  CloseHandle(hEvent);
  AdClose(hDeviceHandle);
  exit(0);
}
   
for (count = 0; count < 10; count++) {
  WaitForSingleObject(hEvent, 5000);
   
// Retrieve the sampling data.
ulSmplNum = 1024;
AdGetSamplingData(hDeviceHandle, &wSmplData[0][0], &ulSmplNum);
if(nRet != AD_ERROR_SUCCESS){
  printf("Error AdGetSamplingData : %08x\n", nRet);
  CloseHandle(hEvent);
  AdClose(hDeviceHandle);
  exit(0);
}
   
for ( i = 0; i < 1024; i++ ) {
  fprintf( fp, "%04d: %04Xh, %04Xh\n", i+1, wSmplData[i][0], wSmplData[i][1];);
}
     
ResetEvent(hEvent);
}
    
// Stop the sampling.
AdStopSampling(hDeviceHandle);
if(nRet != AD_ERROR_SUCCESS){
  printf("Error AdStopSampling : %08x\n", nRet);
  CloseHandle(hEvent);
  AdClose(hDeviceHandle);
exit(0);
}
   
ResetEvent(hEvent);
CloseHandle(hEvent);
   
// Close the file.
fclose(fp);
   
// Clean up and exit.
AdClose(hDeviceHandle);
}

Unload Me

//======================================================
//  Sample program for the Interface analog input 
//  boards/cards 
//   under Windows 7/Server 2008/Vista/XP/Server 2003/ 
//   2000/NT/Me/98/95 
//
//     Copyright 2000, 2011 Interface Corporation.
//     All rights reserved.
//
//  Description
// The infinite sampling is performed asynchronously. 
// An event is occurred by a multiple of 1024 and 
// retrieve the data.
//   And save the retrieved data to the file.
//   Therefore, stop the sampling when writing 10 times //   to the file.
//=======================================================
var
hDeviceHandle: THandle; // Device handle
dwSmplData: array[0..1024, 0..1] of WORD; // Sampling data
SmplConfig: ADSMPLREQ;  // Sampling conditions
szDeviceName: array[0..15] of Char; // Device name
ulSmplNum: Cardinal;  // Number of samples
hEvent: DWORD; // Event handle
F: textFile; // file handle
i: Word;
count: Word;
       
begin
// Initialize the FBIAD1.
StrCopy(szDeviceName, 'FBIAD1');
hDeviceHandle := AdOpen(@szDeviceName[0]);
if hDeviceHandle = INVALID_HANDLE_VALUE then begin
MessageDlg("FBIAD1 is not available.', mtInformation, [mbOK], 0); 
Close;    // Form is closed.;
end;
      
// Configures event handling of the board/card.
hEvent := CreateEvent(nil,TRUE,FALSE,nil);
nRet := AdSetBoardConfig(ghDeviceHandle, hEvent, nil,0);
if nRet <> 0 then begin
  CloseHandle(hEvent);
  MessageDlg('Error code: ' + IntToHex(nRet,8) + 'h', mtError, [mbOk], 0); 
  AdClose(hDeviceHandle);
  Close;
end;
      
// Retrieve the sampling conditions.
nRet = AdGetSamplingConfig(hDeviceHandle, SmplConfig);
if nRet <> 0 then begin
  CloseHandle(hEvent);
  MessageDlg('Error code: ' + IntToHex(nRet,8) + 'h', mtError, [mbOk], 0);
  AdClose(hDeviceHandle);
  Close;
end;
   
// Set 2048 to the number of samples.
// Set 1024 to the number of events.
// Set infinite sampling to trigger mode.
// Channel 1 and channel 2 is set to the sampling channel.
SmplConfig.ulSmplNum := 2048;
SmplConfig.ulSmplEventNum := 1024;
SmplConfig.ulTrigMode := AD_ETERNITY;
SmplConfig.ulChCount := 2;
SmplConfig.SmplChReq[0].ulChNo := 1;
SmplConfig.SmplChReq[0].ulRange := AD_5V;
SmplConfig.SmplChReq[1].ulChNo := 2;
SmplConfig.SmplChReq[1].ulRange := AD_5V;
nRet = AdSetSamplingConfig(hDeviceHandle, SmplConfig);
if nRet <> 0 then begin
  CloseHandle(hEvent);
  MessageDlg('Error code: ' + IntToHex(nRet,8) + 'h', mtError, [mbOk], 0);
  AdClose(hDeviceHandle);
  Close;
end;
   
AssignFile(F, 'data.dat');
Rewrite(F);
   
// Start the sampling as an overlapped operation.
nRet = AdStartSampling(hDeviceHandle, FLAG_ASYNC);
if nRet <> 0 then begin
  CloseHandle(hEvent);
  MessageDlg('Error code: ' + IntToHex(nRet,8) + 'h', mtError, [mbOk], 0);
  AdClose(hDeviceHandle);
  Close;
end;
   
for count := 0 to 10 do begin
  WaitForSingleObject(hEvent, 5000);
   
// Retrieve the sampling data.
ulSmplNum := 1024;
nRet = AdGetSamplingData(hDeviceHandle, @dwSmplData[0,0], ulSmplNum);
if nRet <> 0 then begin
  CloseHandle(hEvent);
  MessageDlg('Error code: ' + IntToHex(nRet,8) + 'h', mtError, [mbOk], 0);
  AdClose(hDeviceHandle);
  Close;
end;
   
for i := 0 to 1023 do begin
Writeln(F, Format('%.4d', [i + 1]) + ': ' + IntToHex(PWordArray(dwSmplData[i,0])) + 'h, ' + IntToHex(PWordArray(dwSmplData[i,1]))) + 'h'); 
end;
   
ResetEvent(hEvent);
end;
   
ResetEvent(hEvent);
   
// Stop the sampling.
nRet = AdStopSampling(hDeviceHandle);
if nRet <> 0 then begin
  CloseHandle(hEvent);
  MessageDlg('Error code: ' + IntToHex(nRet,8) + 'h', mtError, [mbOk], 0);
  AdClose(hDeviceHandle);
  Close;
end;
   
closeFile(F);
   
// Clean up and exit.
AdClose(hDeviceHandle);
end;
